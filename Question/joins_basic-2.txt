-- Create Database
CREATE DATABASE cts_practice;
USE cts_practice;

-- Employees Table
CREATE TABLE Employees (
    EmpID INT PRIMARY KEY,
    EmpName VARCHAR(50),
    DeptID INT,
    Salary DECIMAL(10,2),
    JoinDate DATE
);

-- Departments Table
CREATE TABLE Departments (
    DeptID INT PRIMARY KEY,
    DeptName VARCHAR(50),
    Location VARCHAR(50)
);

-- Projects Table
CREATE TABLE Projects (
    ProjID INT PRIMARY KEY,
    ProjName VARCHAR(50),
    DeptID INT,
    StartDate DATE
);
